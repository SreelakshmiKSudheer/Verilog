//design
module norGate(A,B,C);
  input A,B;
  output C;
  nor(C,A,B);
endmodule

// test bench
module TB;
  reg a,b;
  wire c;
  
  norGate uut(a,b,c);
  
  initial begin
    a = 1'b0; b = 1'b0;
    #100 a = 1'b0; b = 1'b1;
    #100 a = 1'b1; b = 1'b0;
    #100 a = 1'b1; b = 1'b1;
  end
  
  initial begin 
    $monitor($time," a = %b b = %b c = %b",a,b,c);
  end
endmodule
